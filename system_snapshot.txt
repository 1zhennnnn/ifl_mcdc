class IFLOrchestrator:

    def __init__(

        self.parser = ASTParser()
        self.engine = MCDCCoverageEngine()
        self.analyzer = GapAnalyzer()

    def run(self, source_path: str | Path) -> IFLResult:

        甇仿?嚗?          1. Layer 1 閫????蝣?          2. ?ａ?瘜典 & ??頛
          3. ?瑁? 3 ?璈?憪葫閰行?靘?          4. 撱箇??? MCDCMatrix
          5. IFL 餈凋誨餈游?嚗?單?????嚗?          6. 蝯?銝血???IFLResult

        Args:

        Returns:

        # ?? 甇仿? 1嚗圾????
        decision_nodes = self.parser.parse_file(str(source_path))
        if not decision_nodes:

        self._inject_probes(instrumented_module, log)

        test_suite: list[dict[str, object]] = []
        for _ in range(3):

            gaps = self.analyzer.analyze(matrix)

                (g for g in gaps if g.condition_id not in self._infeasible),
                None,
            )
            if gap is None:
                break  # ??擗撩???銝銵?
            # 甇仿? 2嚗3 ??蝝? 峖_gap嚗??啣銵圾蝛粹? 峏嚗ound_specs嚗?            try:
                smt_result = self.smt.synthesize(
                    dn, gap, self.config.domain_types
                )
            except (Z3TimeoutError, Z3UNSATError):
                self._infeasible.add(gap.condition_id)
                loss_history.append(matrix.compute_loss())
                continue

            if not smt_result.satisfiable:
                self._infeasible.add(gap.condition_id)
                loss_history.append(matrix.compute_loss())
                continue

            # 甇仿? 3嚗? 峏 頧??箄?嗉?閮?內嚗??LLM ?Ｙ?皜祈岫獢?
            # LLMSampler ?折?? JSON 閫??憭望???DomainValidator 撽?嚗?憭?閰?MAX_RETRIES 甈∴?
            p_prompt = self.prompt.build(
                dn,
                gap,
                smt_result.bound_specs or [],
                self.config.func_signature,
                self.config.domain_context,

            # 甇仿? 4嚗銵?True ?湛?AcceptanceGate 撽?

                true_id = self._run_test(instrumented_module, new_case, log)

                    test_suite.append(
                        {**new_case, "__test_id": true_id, "__source": "llm"}

            except LLMSamplingError:
                pass

            # 甇仿? 5嚗alse ?湛?鋆?嚗???SMT complement_bound_specs 蝝?嚗???LLM ??
            # ?拙????靽??血?璇辣?潔??湛?瘨??True ?湧蝵拙??渲????????            if smt_result.complement_bound_specs:

                    flip_direction="T2F" if gap.flip_direction == "F2T" else "F2T",
                )
                comp_prompt = self.prompt.build(
                    dn, comp_gap,
                    smt_result.complement_bound_specs,
                    self.config.func_signature,
                    self.config.domain_context,
                )
                try:
                    false_case, _ = self.sampler.sample(comp_prompt)
                    false_id = self._run_test(instrumented_module, false_case, log)
                    if self.gate.evaluate(matrix, log, false_id):

                            {**false_case, "__test_id": false_id, "__source": "llm"}
                        )
                except LLMSamplingError:
                    pass

            loss_history.append(matrix.compute_loss())

        # ?? 甇仿? 6嚗?鋆?????
        total_tokens = sum(
            cast(int, e.get("est_tokens", 0))
            for e in self.sampler.token_log

        return IFLResult(
            converged=matrix.compute_loss() == 0,
            final_coverage=matrix.coverage_ratio,
            test_suite=test_suite,
            iteration_count=iteration,
            total_tokens=total_tokens,
            infeasible_paths=list(self._infeasible),
            loss_history=loss_history,
        )

        return test_id

    @staticmethod
    def _load_from_string(source: str, module_name: str) -> types.ModuleType:
        """??蝺刻陌銝西??亙?銵冽?芋蝯?""

    @staticmethod
    def _generate_random_test(
        dn: DecisionNode,
        domain_types: dict[str, str],
        domain_bounds: dict[str, list[int]] | None = None,
    ) -> dict[str, object]:
        """?箸捱蝑?暺?????貊??璈潦?
        int ??randint(lo, hi)嚗?????domain_bounds嚗?∟身摰? [0, 130]??        bool ??choice([True, False])
        """

        seen: set[str] = set()
        for cond in dn.condition_set.conditions:
            for var_name in cond.var_names:
                if var_name in seen:
                    continue
                seen.add(var_name)
                var_type = domain_types.get(var_name, "int")
                if var_type == "bool":
                    test[var_name] = random.choice([True, False])
                else:
                    bounds = (domain_bounds or {}).get(var_name, [0, 130])
                    test[var_name] = random.randint(bounds[0], bounds[1])
        return test

class OpenAIBackend(LLMBackend):
    """OpenAI ChatCompletion 敺垢??""

    def __init__(
        self,
        model: str = "gpt-4o",
        api_key: str = "",
        temperature: float = 0.3,
    ) -> None:
        self.model = model
        self.api_key = api_key
        self.temperature = temperature

    def complete(self, prompt: str, max_tokens: int = 512) -> str:
        """?澆 OpenAI API??摰? openai 憟辣銝西身摰?OPENAI_API_KEY??""
        import openai
        client = openai.OpenAI(api_key=self.api_key or None)
        resp = client.chat.completions.create(
            model=self.model,
            temperature=self.temperature,
            max_tokens=max_tokens,
            messages=[{"role": "user", "content": prompt}],
        )
        return resp.choices[0].message.content or ""

class LLMSampler:
    """?澆 LLM 敺垢嚗?閰西圾??JSON ????霅?閮? token 瘨?

    MAX_RETRIES: int = 3

    def __init__(
        self,
        backend: LLMBackend,
        validator: DomainValidator,
        retry_delay: float = 2.0,
    ) -> None:
        self.backend = backend
        self.validator = validator
        self.retry_delay = retry_delay
        self.token_log: list[dict[str, object]] = []

    def sample(self, prompt: str) -> tuple[dict[str, object], ValidationResult]:
        """?憭?閰?MAX_RETRIES 甈∴??蝚砌??? DomainValidator ??(dict, ValidationResult)??
        ?岫閫貊璇辣嚗SON 閫??憭望??雯頝舫隤扎???霅仃??
        Args:
            prompt: Gap-Guided Prompt??
        Returns:
            (parsed_dict, validation_result)??
        Raises:
            LLMSamplingError: ?券?岫憭望?嚗SON 閫???雯頝舀?撽??航炊嚗?        """
        current_prompt = prompt
        last_error: str = ""

        for attempt in range(1, self.MAX_RETRIES + 1):
            # 蝚?2 甈∟絲嚗?輻?敺?            if attempt > 1:
                time.sleep(self.retry_delay * (attempt - 1))
                current_prompt = self._build_retry_prompt(prompt, last_error)

            t_start = time.time()
            try:
                raw = self.backend.complete(current_prompt)
            except Exception as exc:
                elapsed = time.time() - t_start
                last_error = str(exc)
                self.token_log.append(
                    {
                        "attempt": attempt,
                        "elapsed": elapsed,
                        "est_tokens": 0,
                    }
                )
                continue

            elapsed = time.time() - t_start
            est_tokens = len(raw) // 4  # 蝎隡啁?
            self.token_log.append(
                {
                    "attempt": attempt,
                    "elapsed": elapsed,
                    "est_tokens": est_tokens,
                }
            )

            data, parse_error = self._parse_json(raw)
            if data is None:
                last_error = f"JSON 閫??憭望?嚗parse_error}"
                continue

            val_result = self.validator.validate(json.dumps(data))
            if not val_result.passed:
                last_error = f"??撽?憭望?嚗val_result.to_corrective_prompt()}"
                continue

            return data, val_result

        raise LLMSamplingError(
            f"LLM ?⊥見憭望?嚗歇?岫 {self.MAX_RETRIES} 甈～?敺隤歹?{last_error}"
        )

    @staticmethod
    def _parse_json(raw: str) -> tuple[dict[str, object] | None, str | None]:
        """?岫敺?LLM ??銝剛圾??JSON ?拐辣??
        甇仿?嚗?          1. 蝘駁 ```json ... ``` ??``` ??
          2. json.loads(cleaned)
          3. 憭望? ????regex ??{ ... }嚗e.DOTALL嚗?閰?
        Returns:
            (dict, None) ??嚗?None, error_str) 憭望???        """
        # 蝘駁 markdown code block
        cleaned = re.sub(r"```json\s*", "", raw)
        cleaned = re.sub(r"```\s*", "", cleaned)
        cleaned = cleaned.strip()

        try:
            data = json.loads(cleaned)
            if isinstance(data, dict):
                return data, None
            return None, f"?? JSON ?拐辣嚗???{type(data).__name__}"
        except json.JSONDecodeError:
            pass

        # ??regex ?? { ... }
        match = re.search(r"\{.*\}", raw, re.DOTALL)
        if match:
            try:
                data = json.loads(match.group())
                if isinstance(data, dict):
                    return data, None
            except json.JSONDecodeError as exc:
                return None, str(exc)

        return None, f"?⊥?敺??葉?? JSON ?拐辣嚗raw[:80]!r}"

    @staticmethod
    def _build_retry_prompt(original: str, error: str) -> str:
        """撱箸? JSON 閫??憭望???閰行?蝷箄???""
        return (
            f"銝?甈∩?????隞乩???嚗????嚗n"
            f"{error}\n\n"
            f"隢?頛詨????JSON ?拐辣嚗?閬?隞颱??嗡????n\n"
            f"???瘙?\n"
